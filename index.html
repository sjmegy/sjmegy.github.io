  <!-- ==========================

       APP: Filtering, search, sort. No frameworks.

  <script>

    const $ = (sel, el=document) => el.querySelector(sel);

    const $$ = (sel, el=document) => Array.from(el.querySelectorAll(sel));


    function formatDate(iso){

      const d = new Date(iso + 'T12:00:00');

      if (Number.isNaN(d.getTime())) return iso;

      return d.toLocaleDateString(undefined, {year:'numeric', month:'short'});

    }


    function uniqueTags(items){

      const set = new Set();

      items.forEach(i => (i.tags||[]).forEach(t => set.add(t)));

      return Array.from(set).sort((a,b)=>a.localeCompare(b));

    }


    function applySort(items, how){

      const by = how || 'newest';

      const arr = [...items];

      if (by === 'newest') arr.sort((a,b)=> (b.date||'').localeCompare(a.date||''));

      if (by === 'oldest') arr.sort((a,b)=> (a.date||'').localeCompare(b.date||''));

      if (by === 'title') arr.sort((a,b)=> a.title.localeCompare(b.title));

      if (by === 'outlet') arr.sort((a,b)=> a.outlet.localeCompare(b.outlet));

      return arr;

    }


    function normalize(str){ return (str||'').toLowerCase(); }


    function searchItems(items, q){

      if (!q) return items;

      const s = normalize(q);

      return items.filter(i => [i.title, i.outlet, i.desc, (i.tags||[]).join(' '), i.type, i.date]

        .join(' ').toLowerCase().includes(s));

    }


    function renderFilters(allItems){

      const box = $('#filters');

      box.innerHTML = '';

      const tags = uniqueTags(allItems);

      const urlTag = new URL(location).searchParams.get('tag');

      const active = new Set(urlTag ? [urlTag] : []);


      const add = (label) => {

        const b = document.createElement('button');

        b.className = 'chip' + (active.has(label) ? ' active' : '');

        b.textContent = label;

        b.onclick = () => {

          if (active.has(label)) active.delete(label); else active.add(label);

          renderList(allItems, {tags: Array.from(active)});

          // update URL param for shareable filter

          const url = new URL(location);

          url.searchParams.delete('tag');

          if (active.size === 1) url.searchParams.set('tag', Array.from(active)[0]);

          history.replaceState({}, '', url);

          renderFilters(allItems);

        };

        box.appendChild(b);

      };


      if (tags.length) add('All');

      tags.forEach(add);


      // Make "All" behave like clearing

      const allBtn = Array.from(box.children).find(btn => btn.textContent === 'All');

      if (allBtn){

        if (active.size === 0) allBtn.classList.add('active');

        allBtn.onclick = () => { active.clear(); renderList(allItems, {tags: []}); history.replaceState({}, '', new URL(location).toString().split('?')[0]); renderFilters(allItems); };

      }

    }


    function renderList(allItems, opts={}){

      const q = $('#q').value.trim();

      const sort = $('#sort').value;

      const tags = new Set(opts.tags || []);


      let items = searchItems(allItems, q);

      if (tags.size) items = items.filter(i => (i.tags||[]).some(t => tags.has(t)));

      items = applySort(items, sort);


      const list = $('#list');

      list.innerHTML = '';


      if (!items.length){

        list.innerHTML = `<div class="card"><h3>No results</h3><p class="desc">Try removing filters or searching different keywords.</p></div>`;

        return;

      }


      items.forEach(i => {

        const el = document.createElement('article');

        el.className = 'card';

        el.innerHTML = `

          <h3>${i.title}</h3>

          <div class="meta">${i.outlet ? i.outlet + ' • ' : ''}${i.date ? formatDate(i.date) : ''}</div>

          <p class="desc">${i.desc || ''}</p>

          <div class="tags">${(i.tags||[]).map(t=>`<span class="tag">${t}</span>`).join('')}</div>

          <a class="primary" href="${i.url}" target="_blank" rel="noopener">Read</a>

        `;

        list.appendChild(el);

      });

    }

    }
